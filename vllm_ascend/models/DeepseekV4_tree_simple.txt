# ───────────────────────────────────────────────────────────────────────
# AscendDSABackend 前向过程详解
# ───────────────────────────────────────────────────────────────────────

├── AscendDSABackend.forward()
│   ├── 1. 输入预处理
│   │   ├── Query路径: hidden_states → wq_a → q_norm → wq_b
│   │   ├── KV路径: hidden_states → wkv → kv_norm
│   │   └── 旋转位置编码: rotary_emb.apply_rotary_emb()
│   ├── 2. 压缩器处理 (compress_ratio > 1)
│   │   └── compressor.forward(hidden_states, positions)
│   │       ├── KV压缩: ape参数 + wkv投影
│   │       ├── 压缩KV缓存更新
│   │       └── 返回压缩后的KV
│   ├── 3. 索引器处理 (compress_ratio == 4)
│   │   └── indexer.forward(hidden_states, qr, positions, rotary_emb)
│   │       ├── Query低秩投影: wq_b
│   │       ├── 索引选择: npu_lightning_indexer
│   │       ├── TopK稀疏索引
│   │       └── 返回稀疏注意力索引
│   ├── 4. 稀疏注意力计算
│   │   └── torch.ops._C_ascend.npu_sparse_flash_attention()
│   │       ├── query: 投影后的Query
│   │       ├── key: 压缩/原始Key
│   │       ├── value: 压缩/原始Value
│   │       ├── sparse_indices: 索引器选择的稀疏索引
│   │       └── attention_sink: 注意力sink参数
│   ├── 5. 输出投影
│   │   ├── wo_a: 组间投影
│   │   └── wo_b: 输出投影
│   └── 6. KV缓存管理
│       ├── 压缩KV缓存更新 (compress_ratio > 1)
│       ├── 索引器KV缓存更新 (compress_ratio == 4)
│       └── SWA缓存更新 (compress_ratio <= 1)

# ───────────────────────────────────────────────────────────────────────
# DeepseekV4Model 架构
# ───────────────────────────────────────────────────────────────────────

├── DeepseekV4Model                                                # vllm-ascend\vllm_ascend\models\deepseek_v4.py#L870
│   ├── ═══════════════
│   │   1、初始化过程
│   │   ═══════════════
│   ├── self.embed_tokens (VocabParallelEmbedding)                 # 词嵌入层
│   ├── self.layers[]: DeepseekV2DecoderLayer × N                  # 解码器层堆叠
│   ├── self.norm (RMSNorm)                                        # 最终归一化
│   ├── self.hc_mult: int                                          # HC倍数
│   ├── self.hc_eps: float                                         # HC epsilon
│   ├── self.hc_head_fn (nn.Parameter)                             # HC头函数
│   ├── self.hc_head_base (nn.Parameter)                           # HC头基
│   ├── self.hc_head_scale (nn.Parameter)                          # HC头缩放
│   ├── self.topk_indices_buffer (torch.Tensor)                    # TopK索引缓冲区
│   └── self._mtp_hidden_buffer (torch.Tensor)                     # MTP隐藏状态缓冲区
│   ├── ═══════════════
│   │   2、forward过程
│   │   ═══════════════
│   └── forward(input_ids, positions, intermediate_tensors, inputs_embeds)
│       ├── 输入嵌入
│       │   ├── if inputs_embeds is not None: hidden_states = inputs_embeds
│       │   └── else: hidden_states = embed_input_ids(input_ids)
│       ├── Llama-4缩放计算 (如果启用)
│       │   └── llama_4_scaling = _get_llama_4_scaling()
│       ├── HC扩展 (如果是第一PP阶段)
│       │   └── hidden_states = hidden_states.unsqueeze(1).repeat(1, hc_mult, 1)
│       ├── 层前向传播
│       │   └── for layer in layers:
│       │       └── hidden_states, residual = layer(positions, hidden_states, residual, llama_4_scaling)
│       ├── MTP隐藏状态保存
│       │   ├── if FlashComm1启用: tensor_model_parallel_all_gather()
│       │   └── self._mtp_hidden_buffer[:num_tokens].copy_(hidden_states)
│       ├── HC头处理 (如果是最后PP阶段)
│       │   └── hidden_states = hc_head(hidden_states, hc_head_fn, hc_head_scale, hc_head_base)
│       │       ├── 归一化: rsqrt(x.square().mean(-1, keepdim=True) + norm_eps)
│       │       ├── 混合: linear(x, hc_fn) * rsqrt
│       │       ├── 激活: sigmoid(mixes * hc_scale + hc_base) + hc_eps
│       │       └── 聚合: sum(pre.unsqueeze(-1) * x.view(shape), dim=1)
│       ├── 最终归一化
│       │   └── hidden_states = norm(hidden_states)
│       └── return hidden_states / IntermediateTensors

│   ├── ═══════════════
│   │   2、forward过程
│   │   ═══════════════
│   └── forward(input_ids, positions, hidden_states, inputs_embeds, spec_step_idx)
│       ├── 输入嵌入
│       │   └── if inputs_embeds is None: inputs_embeds = embed_input_ids(input_ids)
│       ├── MTP层选择
│       │   └── current_step_idx = spec_step_idx % num_mtp_layers
│       ├── MTP层前向传播
│       │   └── layers[current_step_idx](input_ids, positions, hidden_states, inputs_embeds, current_step_idx)
│       │       ├── 输入掩码: torch.where(positions == 0, 0, inputs_embeds)
│       │       ├── E归一化: enorm(inputs_embeds)
│       │       ├── H归一化: hnorm(previous_hidden_states)
│       │       ├── 投影融合: e_proj(inputs_embeds) + h_proj(previous_hidden_states)
│       │       ├── MTP块处理: mtp_block(positions, hidden_states, residual)
│       │       └── return hidden_states
│       └── return hidden_states
│   ├── ═══════════════

# ───────────────────────────────────────────────────────────────────────
# 关键特性说明
# ───────────────────────────────────────────────────────────────────────

├── 1. 混合协同处理 (HC - Hybrid Collaboration)
│   ├── hc_pre: 前处理混合协同
│   │   └── torch.ops._C_ascend.npu_hc_pre(x, hc_fn, hc_scale, hc_base, hc_mult, hc_sinkhorn_iters, norm_eps, hc_eps)
│   ├── hc_post: 后处理混合协同
│   │   └── torch.ops._C_ascend.npu_hc_post(x, residual, post, comb)
│   └── hc_head: HC头处理
│       ├── 归一化: rsqrt(x.square().mean(-1, keepdim=True) + norm_eps)
│       ├── 混合: linear(x, hc_fn) * rsqrt
│       ├── 激活: sigmoid(mixes * hc_scale + hc_base) + hc_eps
│       └── 聚合: sum(pre.unsqueeze(-1) * x.view(shape), dim=1)
│
├── 2. 稀疏注意力 (DSA - DeepSeek Sparse Attention)
│   ├── 压缩器 (Compressor): KV压缩 (compress_ratio: 2/4)
│   ├── 索引器 (Indexer): TopK稀疏索引 (compress_ratio == 4)
│   ├── 注意力Sink: attn_sink参数
│   └── NPU算子: npu_sparse_flash_attention
│
├── 3. 混合专家 (MoE - Mixture of Experts)
│   ├── 路由机制: gate线性层 + hash路由 (tid2eid)
│   ├── EPLB: 专家并行负载均衡
│   ├── 共享专家: shared_experts (可选)
│   └── NPU融合算子: moe_gating_top_k_hash
│
├── 4. 多Token预测 (MTP - Multi-Token Prediction)
│   ├── MTP层: DeepSeekMultiTokenPredictorLayer
│   ├── MTP块: DeepseekV2DecoderLayer
│   ├── 共享头: SharedHead
│   └── HC头: hc_head处理
│
├── 5. NPU优化
│   ├── NPU融合算子: torch.ops._C_ascend.*
│   ├── NPU稀疏注意力: npu_sparse_flash_attention
│   ├── NPU MoE路由: moe_gating_top_k_hash
│   ├── NPU HC处理: npu_hc_pre/npu_hc_post
│   └── NPU索引器: npu_lightning_indexer
│
└── 6. 并行策略
    ├── TP (Tensor Parallel): 张量并行
    ├── PP (Pipeline Parallel): 流水线并行
    ├── EP (Expert Parallel): 专家并行
    ├── DP (Data Parallel): 数据并行
    ├── MC2 (MoE Communication): MoE通信
    └── SP (Sequence Parallel): 序列并行
